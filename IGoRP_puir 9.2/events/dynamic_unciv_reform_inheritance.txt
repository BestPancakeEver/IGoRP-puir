#land_reform 1-giver
country_event = {
	id = 820000
	title = "EVTNAME820000"
	desc = "EVTDESC820000"

	picture = "Rebellion"

	trigger = { 
		OR = {
			has_country_flag = released_colonial_MLY
			has_country_flag = released_colonial_THI
			has_country_flag = released_colonial_ICU
		}
		AND = {
		    practical_steam_engine = 1
		    clean_coal = 1
		}
		#Failsafe against reform reversion
		NOT = {
		    AND = {
		       market_structure = 1
		       high_n_low_pressure_steam_engines = 1
		    }
		}
		OR = {
		   MLY = { in_sphere = THIS }
		   THI = { in_sphere = THIS }
		   ICU = { in_sphere = THIS }
		}
		NOT = { has_country_flag = land_reform_1 }
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
	    name = "Lets share our technology."
	    random_country = {
			limit = {
				AND = {
				    OR = {
				       tag = MLY
					   tag = THI
					   tag = ICU
				    }
					in_sphere = THIS
				}
				
			}
			country_event = 820020
		}
		set_country_flag = land_reform_1
	}
}
