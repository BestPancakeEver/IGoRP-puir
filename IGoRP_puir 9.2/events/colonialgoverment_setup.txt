	desc = "EVTDESC810002"

	picture = "meiji_court"

	trigger = {
	    tag = MLY
		ai = no
		NOT = { has_country_flag = MLY_noai }
		
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "A great achivement indeed!"
		set_country_flag = MLY_noai
		any_owned = {
					add_core = MLY
				    remove_core = SIA
				    remove_core = JOH
				    remove_core = BRU
				    remove_core = ATJ
				    remove_core = DJA
				    remove_core = KLM
				    remove_core = BAL
				    remove_core = PHI
				    remove_core = PHL
					remove_core = TRA
		}
		add_country_modifier = {
			name = colonial_goverment_support

		}

		random_country = {
			limit = {
				has_country_flag = released_colonial_MLY
			}
			country_event = 809999
		}
		
		
    }
}
